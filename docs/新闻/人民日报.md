<script>
(function() {
    // === 1. DOM 元素获取 ===
    const iframe = document.getElementById('daily-paper-iframe');
    const urlText = document.getElementById('current-date-text');
    const datePicker = document.getElementById('date-picker');
    const btnPrev = document.getElementById('btn-prev');
    const btnNext = document.getElementById('btn-next');

    // === 2. 初始化时间 ===
    // 获取“今天”的 Date 对象，并将时间归零 (00:00:00)，确保只比较日期
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    // 当前正在查看的日期
    let currentDate = new Date(today);

    // === 3. 核心逻辑函数 ===


    function getDateInfo(date) {
        const y = date.getFullYear();
        const m = (date.getMonth() + 1).toString().padStart(2, '0');
        const d = date.getDate().toString().padStart(2, '0');

        return {
        dateObj: date,
        iso: `${y}-${m}-${d}`, // 格式: 2025-01-01 (用于 input)
        urlPath: `${y}${m}/${d}`, // 格式: 202501/01 (用于 URL)
        display: `${y}年${m}月${d}日` // 格式: 中文显示
        };
    }


    function loadPaper(targetDate) {
    // 确保传入的时间归零
        targetDate.setHours(0, 0, 0, 0);
        currentDate = new Date(targetDate);

        const info = getDateInfo(currentDate);

        // 1. 生成 URL
        // 官网结构: .../layout/YYYYMM/DD/node_01.html
        const baseUrl = "https://paper.people.com.cn/rmrb/pc/layout";
        const fullUrl = `${baseUrl}/${info.urlPath}/node_01.html`;

        // 2. 更新 Iframe
        if (iframe) {
            iframe.src = fullUrl;
        }

        // 3. 更新地址栏文字
        if (urlText) {
            urlText.textContent = `人民日报 / ${info.display} (第01版)`;
            urlText.onclick = () => window.open(fullUrl, '_blank');
        }

        // 4. 更新日期选择器的值
            if (datePicker) {
            datePicker.value = info.iso;
        }

        // 5. [关键] 控制“后一天”按钮状态
        // 如果当前日期 >= 今天，禁用按钮
        if (btnNext) {
            if (currentDate.getTime() >= today.getTime()) {
                btnNext.disabled = true;
                btnNext.title = "已是最新日期";
            } else {
                btnNext.disabled = false;
                btnNext.title = "后一天";
            }
        }
    }

    // === 4. 事件绑定 ===

    // 点击【前一天】
    btnPrev.addEventListener('click', function() {
        const prevDate = new Date(currentDate);
        prevDate.setDate(prevDate.getDate() - 1);
        loadPaper(prevDate);
    });

    // 点击【后一天】
    btnNext.addEventListener('click', function() {
    // 逻辑校验：防止手动修改 DOM 属性后点击
    if (currentDate.getTime() >= today.getTime()) return;

        const nextDate = new Date(currentDate);
        nextDate.setDate(nextDate.getDate() + 1);
        loadPaper(nextDate);
    });

    // 【日期选择器】变更
    datePicker.addEventListener('change', function(e) {
        if (!e.target.value) return;

            // 解析输入值 YYYY-MM-DD
            // 注意：直接 new Date("YYYY-MM-DD") 在不同时区可能解析有误，
            // 建议分割字符串手动构造 Date
            const parts = e.target.value.split('-');
            const selectedDate = new Date(parts[0], parts[1] - 1, parts[2]);

        // 边界检查：如果选了未来日期，强制回退到今天
        if (selectedDate.getTime() > today.getTime()) {
            loadPaper(today);
        } else {
            loadPaper(selectedDate);
        }
    });

    // === 5. 初始化设置 ===

    // 设置日期选择器的 max 属性，原生禁止选择未来日期
    datePicker.max = getDateInfo(today).iso;

    // 默认加载今天
    loadPaper(today);

})();

</script>
