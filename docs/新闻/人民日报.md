<script>
(function() {
    // 1. 获取 DOM 元素
    var iframe = document.getElementById('daily-paper-iframe');
    var urlText = document.getElementById('current-date-text');

    // 2. 获取当前系统时间
    var now = new Date();
    var year = now.getFullYear();
    
    // 月份需要 +1，且如果小于10需要补0 (例如 5 -> 05)
    var month = (now.getMonth() + 1).toString().padStart(2, '0');
    
    // 日期如果小于10需要补0
    var day = now.getDate().toString().padStart(2, '0');

    // 3. 拼接 URL
    // 官网格式：.../layout/202512/17/node_01.html
    // 即：.../layout/YYYYMM/DD/node_01.html
    var baseUrl = "https://paper.people.com.cn/rmrb/pc/layout";
    var datePath = year + month + "/" + day;
    var fullUrl = baseUrl + "/" + datePath + "/node_01.html";

    // 4. 更新页面内容
    if (iframe) {
        iframe.src = fullUrl;
    }

    if (urlText) {
        // 地址栏显示友好一点的信息
        urlText.textContent = "人民日报 / " + year + "-" + month + "-" + day + " (今日头版)";
        
        // 点击地址栏可以跳转到官网（防止 iframe 加载失败时无法查看）
        urlText.onclick = function() {
            window.open(fullUrl, '_blank');
        };
    }
})();
</script>
